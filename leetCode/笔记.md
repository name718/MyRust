# 求余、求模、保留小数
```rust
fn main() {
    // 求余
    let a = 17;
    let b = 5;
    let remainder = a % b;
    println!("求余: {} % {} = {}", a, b, remainder);

    // 求模 (通常与求余相同)
    let modulus = ((a % b) + b) % b; // 确保结果为正
    println!("求模: {} mod {} = {}", a, b, modulus);

    // 保留小数
    let c = 5.12345;
    let d = 2.0;
    let result = c / d;
    println!("未保留小数: {}", result);

    // 保留小数 (例如保留两位小数)
    let rounded_result = (result * 100.0).round() / 100.0;
    println!("保留两位小数: {:.2}", rounded_result);
}

```
# 向下取整，向上取整
```rust
fn main() {
    let x = 5.67;

    // 向下取整 (floor)
    let down = x.floor();
    println!("向下取整: {} -> {}", x, down);

    // 向上取整 (ceil)
    let up = x.ceil();
    println!("向上取整: {} -> {}", x, up);
}

```
